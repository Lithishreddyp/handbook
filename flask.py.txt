import json
import pytest
from app import create_app, db
from app.models.task import Task
from app.models.comment import Comment

@pytest.fixture
def app():
    app = create_app("testing")  # adapt to your create_app signature
    with app.app_context():
        db.create_all()
        yield app
        db.session.remove()
        db.drop_all()

@pytest.fixture
def client(app):
    return app.test_client()

@pytest.fixture
def sample_task(app):
    t = Task(title="Sample", description="x")  # adjust fields to your Task model
    db.session.add(t)
    db.session.commit()
    return t

def test_create_comment(client, sample_task):
    resp = client.post(f"/api/tasks/{sample_task.id}/comments",
                       json={"author": "Bob", "body": "first comment"})
    assert resp.status_code == 201
    data = resp.get_json()
    assert data["author"] == "Bob"
    assert data["body"] == "first comment"
    assert "id" in data

def test_list_comments_empty(client, sample_task):
    resp = client.get(f"/api/tasks/{sample_task.id}/comments")
    assert resp.status_code == 200
    assert resp.get_json() == []

def test_get_comment_not_found(client, sample_task):
    resp = client.get(f"/api/tasks/{sample_task.id}/comments/9999")
    assert resp.status_code == 404

def test_update_comment(client, sample_task):
    # create
    r = client.post(f"/api/tasks/{sample_task.id}/comments", json={"author": "A", "body": "old"})
    comment_id = r.get_json()["id"]
    # update
    r2 = client.put(f"/api/tasks/{sample_task.id}/comments/{comment_id}", json={"body": "new"})
    assert r2.status_code == 200
    assert r2.get_json()["body"] == "new"

def test_delete_comment(client, sample_task):
    r = client.post(f"/api/tasks/{sample_task.id}/comments", json={"author": "A", "body": "bye"})
    comment_id = r.get_json()["id"]
    r2 = client.delete(f"/api/tasks/{sample_task.id}/comments/{comment_id}")
    assert r2.status_code == 204
    # confirm gone
    r3 = client.get(f"/api/tasks/{sample_task.id}/comments/{comment_id}")
    assert r3.status_code == 404

def test_create_comment_invalid(client, sample_task):
    r = client.post(f"/api/tasks/{sample_task.id}/comments", json={"author": "A", "body": ""})
    assert r.status_code == 400
